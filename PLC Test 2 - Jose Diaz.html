def checkBool(end):
    parenCount = 0
    curlyCount = 0
    startBool = False
    totalBool = False
    for char in end:
        if(char == '('):
            parenCount += 1
        elif(char == ')'):
            parenCount -= 1
        elif(char == '{'):
            curlyCount += 1
        elif(char == '}'):
            curlyCount -= 1
        elif(char == '>' or char == '<' or char == '='):

        if(startBool and char == '='):
            totalBool = True
            startBool = False
    return (startBool or totalBool) and (curlyCount == 0) and (parenCount == 0)
